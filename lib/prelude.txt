# Functions.

# http://en.wikipedia.org/wiki/Permutation
function permutations(n:int, r:int):int {
  return n!/(n-r)!;
}

# http://en.wikipedia.org/wiki/Combination
function combinations(n:int, k:int):int {
  return n!/(k! * (n-k)!);
}
